const express = require('express'); const cors = require('cors'); const bodyParser = require('body-parser'); const app = express(); const PORT = 5000; app.use(cors()); app.use(bodyParser.json()); let contacts = []; // temporary memory database // Get all contacts app.get('/contacts', (req, res) => { res.json(contacts); }); // Add new contact app.post('/contacts', (req, res) => { const newContact = req.body; contacts.push(newContact); res.status(201).json(newContact); }); // Delete contact by name app.delete('/contacts/:name', (req, res) => { const name = req.params.name; contacts = contacts.filter(contact => contact.name !== name); res.status(200).json({ message: 'Contact deleted' }); }); app.listen(PORT, () => { console.log(`Server running on http://localhost:${PORT}`); });
